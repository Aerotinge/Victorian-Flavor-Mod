# Root = Colony state
on_colony_created = {
	effect = {
		owner = {
			if = {
				limit = {
					has_journal_entry = je_tutorial_colonize_state
					je:je_tutorial_colonize_state = {
						NOT = {
							has_variable = tut_colony_target
						}
					}
				}
				je:je_tutorial_colonize_state = {
					set_variable = {
						name = tut_colony_target
						value = root
					}
				}
			}
		}
	}
}
