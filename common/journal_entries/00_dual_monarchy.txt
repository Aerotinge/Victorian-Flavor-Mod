je_dual_monarchy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	complete = {
		cu:hungarian = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		c:AUS = {	
			c:HUN.relations:ROOT > relations_threshold:amicable
		}
		c:HUN = {
			has_attitude = {
				who = c:AUS
				attitude = loyal
			}	
		} 
	}

	fail = {
		c:AUS = {	
			c:HUN.relations:ROOT < relations_threshold:cold
		}
		c:HUN = {
			has_attitude = {
				who = c:AUS
				attitude = rebellious
			}	
		} 
	 }
	on_fail = {
		c:HUN = {	
			create_diplomatic_play = {
				name = confederation_war
				target_country = c:AUS
				type = dp_independence
			}
		}

	}

	on_complete = {
		trigger_event = {
			id = dual_monarchy.1
		}
	}

	invalid = {
		OR = {
			NOT = {
				c:HUN = {
					is_subject_of = c:AUS
				}
			}
			NOT = {
				this = c:AUS
			}
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}
