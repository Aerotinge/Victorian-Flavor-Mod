je_hanja_issue = {
	icon = "gfx/interface/icons/event_icons/event_academia.dds"

	complete = {
		OR = {
			AND = {
				any_scope_building = {
					is_building_type = building_university
					occupancy > 0.9
					level >= 5
				}
				institution_investment_level = {
					institution = institution_schools 
					value >= 2
				}
			}
			ig:ig_intelligentsia = {
				is_powerful = yes
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = kor.1
		}
	}
}

je_sadae = {
	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	complete = {
		OR = {
			top_overlord = {
				custom_tooltip = {
					text = kor_chi_relations_tt
					relations:root >= relations_threshold:amicable
				}
				custom_tooltip = {
					text = kor_chi_obligation_tt
					owes_obligation_to = root
				}
			}
			gdp_chi = {
				value = {
					add = c:CHI.GDP
					multiply = 0.2
				}
			}
			custom_tooltip = {
				text = kor_chi_gdp_tt
				GDP >= gdp_chi
			}
		}
	}

	fail = {
		top_overlord = {
			custom_tooltip = {
				text = kor_chi_relations_bad_tt
				relations:root < relations_threshold:neutral
			}
			is_at_war = no
		}
	}

	invalid = {
		AND = {
			NOT = {
				c:CHI = {
					has_diplomatic_pact = {
						who = ROOT
						type = tributary
						is_initiator = yes
					}
				}
			}
			NOT = { is_subject_of = c:CHI }
		}	
	}

	on_complete = {
		trigger_event = {
			id = kor.3
		}
	}

	on_fail = {
		c:CHI = {
			trigger_event = {
				id = kor.4
			}
		}
	}
}

je_silhak_brewing = {
	icon = "gfx/interface/icons/event_icons/event_revolt.dds"

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		custom_tooltip = {
			text = kor_silhak_criteria
		}
	}

	current_value = {
		value = var:silhak_var
	}
	goal_add_value = {
		add = 120
	}
	progressbar = yes

	on_complete = {

	}

	on_weekly_pulse = {
		if = {
			limit = { #while scholars are influential and not being suppressed move closer to revolt
				AND = { 
					ig:ig_intelligentsia = {is_marginal = false}
					ig:ig_intelligentsia = {is_being_suppressed = no}
				}
			}
			change_variable = {
				name = silhak_var
				add = 1
			}
		}
		if = {
			limit = { #move at double speed if they are powerful
				ig:ig_intelligentsia = {is_powerful = true}
			}
			change_variable = {
				name = silhak_var
				add = 1
			}
		}
		if = {
			limit = { #subtract while intelligentsia are marginal
				ig:ig_intelligentsia = {is_marginal = true}
			}
			change_variable = {
				name = silhak_var
				subtract = 1
			}
		}
	}
}
