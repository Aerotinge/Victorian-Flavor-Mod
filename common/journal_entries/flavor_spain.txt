je_carlist_victory = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	complete = {
        custom_tooltip = {
			text = win_civil_war
			hidden_trigger = {
				NOT = { exists = c:SPA }
			}
		}	
       
	}

	on_complete = {
		change_tag = SPA
		set_variable = carlist_spain
		hidden_effect = {
		    c:SPA = {
			    trigger_event  = {
				    id = journal_flavor_spain.2
				        months = 2
			    }
			} 
		}	 
	}

    should_be_pinned_by_default = yes
	weight = 100
}
