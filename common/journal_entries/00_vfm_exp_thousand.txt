je_exp_thousand = {

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	immediate = {
		c:SIC = {
			save_scope_as = sic_scope
			add_journal_entry = {
				type = je_exp_thousand_SIC
			}
		}
		random_country = {
			limit = {
				owns_entire_state_region = STATE_MALTA
			}
			save_scope_as = malta_scope
		}
		set_variable = choose_commander
	}

	complete = {
		any_country = {
			limit = {
				owns_entire_state_region = STATE_MALTA
			}
			relations:root >= 10
		}
		this.gdp > c:SIC.gdp
		c:SIC = {
			is_at_war = no
		}
		is_at_war = no
		in_default = no
		taking_loans = no
		is_unification_candidate = ITA

    	custom_tooltip = {
    		text = exp_thousand_commander
			has_variable = has_thousand_commander
		}
    }

    on_complete = {
		add_journal_entry = {
			type = je_exp_thousand_ongoing
		}
    }

    invalid = {
    	NOT = {
    		exists = c:SIC
    	}
    }

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = exp_thousand_goal
			}
		}
	}

	timeout = 730
	weight = 1000

	should_be_pinned_by_default = yes
}

je_exp_thousand_SIC = {

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	immediate = {
		random_country = {
			limit = {
				has_journal_entry = je_exp_thousand
			}
			save_scope_as = ita_scope
		}
	}

	complete = {
		radical_fraction = {
			value < 0.15
		}
		any_scope_state = {
			turmoil < 0.1
		}	
		any_interest_group = {
			ig_approval >= -5
		}
    }

    on_complete = {
    	set_global_variable = exp_thousand_limited
    }

    fail = {
	    custom_tooltip = {
    		text = exp_thousand_SIC_fail
	    	any_country = {
	    		has_variable = garibaldi_redshirts
	    	}
	    }
    }

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = exp_thousand_SIC_goal
			}
		}
	}
	weight = 1000

	should_be_pinned_by_default = yes
}


je_exp_thousand_ongoing = {

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = expedition_events.20 #posts a notification
			99 = exp_thousand.101 #the landing
			50 = exp_thousand.102 #calatafimi battle
			20 = exp_thousand.103 #population joins us
			10 = exp_thousand.104 #reinforcements
			10 = exp_thousand.105 #milazzo battle
			30 = exp_thousand.106 #reggio battle
			50 = exp_thousand.107 #capture of Naples
			20 = exp_thousand.108 #volturno battle
		}
	}

	immediate = {
		random_scope_character = {
			limit = {
				has_variable = thousand_commander
			}
			save_scope_as = exp_commander
		}
		c:SIC = {
			save_scope_as = sic_scope
		}
		trigger_event = { id = exp_thousand.101 days = 7 }
	}
	complete = {
		custom_tooltip = {
			text = je_expedition_success
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
    }

    on_complete = {
		c:SIC = {
			trigger_event = { id = exp_thousand.3 }
		}
		remove_modifier = modifier_large_expedition_cost
    }

    fail = {
		custom_tooltip = {
			text = je_expedition_fail
			var:expedition_peril >= 7
		}
    }

	on_fail = {
		c:SIC = {
			trigger_event = { id = exp_thousand.5 }
		}
		remove_modifier = modifier_large_expedition_cost
	}

	current_value = {
		value = var:expedition_progress
	}

	goal_add_value = {
		add = 7
	}

    invalid = {
    	NOT = {
    		exists = c:SIC
    	}
    }

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = exp_thousand_ongoing_goal
			}
		}
	}

	progressbar = yes

	weight = 1000

	should_be_pinned_by_default = yes
}
