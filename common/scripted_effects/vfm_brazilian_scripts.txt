	hidden_effect = {
		remove_modifier = soldiers_insatisfied
	}

		}

add_dom_pedro_legitimacy_effect = {
	if = {
		limit = {
			var:dom_pedro_legitimacy <= 3
		}
		ruler = {
			add_modifier = {
				name = emperor_dom_pedro
			}
			add_modifier = {
				name = emperor_dom_pedro_temp
				months = long_modifier_time
			}
		}
		add_modifier = {
			name = regency_ended
			months = long_modifier_time
			is_decaying = yes
		}
	}
	else_if = {
		limit = {
			var:dom_pedro_legitimacy <= 6
		}
		ruler = {
			add_modifier = {
				name = emperor_dom_pedro
			}
			add_modifier = {
				name = emperor_dom_pedro_temp
				months = long_modifier_time
				multiplier = 2
			}
		}
		add_modifier = {
			name = regency_ended
			months = long_modifier_time
			multiplier = 2
			is_decaying = yes
		}
	}
	else_if = {
		limit = {
			var:dom_pedro_legitimacy <= 9 #should be impossible to get as high as 9
		}
		ruler = {
			add_modifier = {
				name = emperor_dom_pedro
				multiplier = 2
			}
			add_modifier = {
				name = emperor_dom_pedro_temp
				months = long_modifier_time
				multiplier = 3
			}
		}
		add_modifier = {
			name = regency_ended
			months = long_modifier_time
			multiplier = 3
			is_decaying = yes
		}
	}
}
