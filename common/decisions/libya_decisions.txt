accept_senussi_order = {

	is_shown = {
		exists = c:AWG
		root = c:AWG
		NOT = { has_variable = senussi_embraced }
	}

	possible = {
		custom_tooltip = {
			text = el_senussi_begins_tt
			date > 1840.1.1
			has_variable = el_senussi_banished
		}
	}

	when_taken = {
		custom_tooltip = {
			text = el_senussi_rises_tt
			set_variable = senussi_embraced
		}
	}

	ai_chance = {
		base = 100
	}
}
