france_invade_CNS = {
	is_shown = {
		exists = c:FRA
		exists = c:CNS
		c:FRA = ROOT
		is_subject = no
		NOT = { has_variable = invaded_constantine_once }
	}
	possible = {
		c:FRA = {
			is_at_war = no
			NOT = {
				has_truce_with = c:CNS
			}
		}
	}
	when_taken = {
		c:FRA = {
			set_variable = {
				name = invaded_constantine_once
			}
			create_diplomatic_play = {
				name = french_algeria_invasion
			
				target_state = s:STATE_CONSTANTINE.region_state:CNS
				
				war = no
				
				type = dp_annex_war

				add_war_goal = {
					holder = c:CNS
					type = conquer_state
					target_state = s:STATE_CONSTANTINE.region_state:FRA
				}
			}
		}
	}

	ai_chance = {
		base = 100
	}
}
