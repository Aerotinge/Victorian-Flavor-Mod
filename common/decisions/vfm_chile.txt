1881_conquest = {

	is_shown = {
		OR = {
			c:CHL = this
			c:ARG = this               
		}
		NOT = { 
            owns_entire_state_region = STATE_PATAGONIA
        }

	    NOT = { has_variable = pehuenche_conquered } 

	}

	    possible = {
		    exists = c:THL
		    c:THL = {
		        is_at_war = no
		        }
		    is_at_war = no
		    institution_investment_level = {
			    institution = institution_colonial_affairs
			    value >= 3
		    }
	        country_rank >= rank_value:minor_power
		    year >= 1881
	        }

	when_taken = {

		hidden_effect = { set_global_variable = 1881_conquest_completed }

		custom_tooltip = {
			text = je_text_1881_conquest
		}               

		hidden_effect = {
			create_diplomatic_play = {
				name = chl_arg_1881_conquest
			
				target_state = s:STATE_PATAGONIA.region_state:THL
				
				war = yes
				
				type = dp_annex_war
			}
		}
        set_global_variable = indigenous_nations_defeated
    }

	fail = {
		NOT = { 
		    exists = c:THL 
		}
		has_global_variable = 1881_conquest_completed	
    }

	ai_chance = {
		base = 100 	
	}	
}

expand_idp = {
	is_shown = {
		exists = c:IDP
		c:IDP = {
			is_subject_of = ROOT
		}
	}
	possible = {
		any_state = {
			owner = ROOT
			region = sr:region_oceania
		}
	}	
	when_taken = {
		custom_tooltip = expand_dip_tt
		if = {
			limit= {
				any_state = {
					owner = ROOT
					region = sr:region_oceania
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					region = sr:region_oceania
				}
				set_state_owner = c:IDP
				set_state_type = unincorporated
			}
		}
	}

	ai_chance = {
		base = 100
	}
}
