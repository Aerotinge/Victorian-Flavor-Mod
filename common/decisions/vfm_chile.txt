la_moneda = {
	
	is_shown = {
		this = c:CHL
		
		NOT = { has_variable = new_president_seat }
	}

	possible = {
			    any_scope_building = {
				    is_building_type = building_construction_sector
				    level >= 1
			    }
			    any_scope_building = {
				    is_building_type = building_government_administration
				    level >= 3
			    }
	        }

	when_taken = {
		trigger_event = {
			id = journal_flavor_chile.10 
			popup = yes
		}
		set_variable = {
			name = new_president_seat
		}
	}
	ai_chance = {
		base = 100 	
	}
}
