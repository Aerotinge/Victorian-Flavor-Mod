namespace = vfm_greece



vfm_greece.1 = {
	type = country_event

	title = vfm_greece.1.t
	desc = vfm_greece.1.d
	flavor = vfm_greece.1.f

	placement = scope:ionian_overlord_capital

	event_image = {
		video = "gfx/europenorthamerica_art_gallery.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	duration = 3

	trigger = {
	}

	immediate = {
		random_primary_culture = {
			save_scope_as = greek_culture_scope
		}
		s:STATE_ATTICA = {
			random_scope_state = {
				limit = {
					owner = {
						c:GRE = THIS
					}
				}
				save_scope_as = attica_state
			}
		}
	}

	option = {
		name = greece.3.a
		default_option = yes
		scope:attica_state = {
		    create_pop = {
				pop_type = academics
				culture = greek
				size = 750			
		    }
		}
	}
}
