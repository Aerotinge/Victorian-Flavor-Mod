vfm_greece.1 = {
	type = country_event

	title = vfm_greece.1.t
	desc = vfm_greece.1.d
	flavor = vfm_greece.1.f

	placement = scope:attica_state

	event_image = {
		video = "gfx/europenorthamerica_art_gallery.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	duration = 3

	trigger = {
	}

	immediate = {
		random_primary_culture = {
			save_scope_as = greek_culture_scope
		}
		s:STATE_ATTICA = {
			random_scope_state = {
				limit = {
					owner = {
						c:GRE = THIS
					}
				}
				save_scope_as = attica_state
			}
		}
	}

	option = {
		name = vfm_greece.1.a
		default_option = yes
		scope:attica_state = {
		    create_pop = {
				pop_type = academics
				culture = greek
				size = 750			
		    }
		}
	}
}

vfm_greece.7 = {
	type = country_event

	title = vfm_greece.7.t
	desc = vfm_greece.7.d
	flavor = vfm_greece.7.f

	placement = scope:ottoman_state

	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	duration = 3

	trigger = {
	    exists = c:GRE
		this = c:GRE
	}

	immediate = {
		c:TUR = {
			save_scope_as = turkey_country
			random_scope_state = {
				limit = {
				    state_region = { 
					    is_homeland = cu:greek
					}
					has_state_trait = state_medeteranean_climate
				}
				save_scope_as = ottoman_state
			}
		}
	}

	option = {
		name = vfm_greece.7.a
		default_option = yes
		add_treasury = -25000
		scope:ottoman_state = {
		    add_radicals = {
				value = 0.15
				culture = cu:greek
			}
			add_devastation = 20
			
			trigger_event = { id = vfm_greece.8 }
		}
		change_infamy = 4
		
		# lower troops by 2 in state
	}
	
	option = {
		name = vfm_greece.7.b
		scope:ottoman_state = {
		    add_radicals = {
				value = 0.1
				culture = cu:greek
			}
			add_devastation = 10
			trigger_event = { id = vfm_greece.9 }
		}
		add_radicals = {
			pop_type = officers
			value = small_radicals
		}
		
		# lower troops by 1 in state
	}
}
