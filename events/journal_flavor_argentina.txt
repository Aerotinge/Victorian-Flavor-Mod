journal_flavor_argentina.1 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_argentina.1.t
	desc = journal_flavor_argentina.1.d
	flavor = journal_flavor_argentina.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_military_parade.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	duration = 1

	immediate = {
		if={
			limit = {
				NOT = { exists = c:ARG }
			}
			change_tag = ARG
		}	
	}

	option = {
		name = journal_flavor_argentina.1.a
		default_option = yes

		if= {
			limit = {
				exists = c:GNI
			}
			s:STATE_CHACO = {
				add_claim = ROOT
			}
			s:STATE_SANTA_FE = {
				add_claim = ROOT
			}
		}
		if= {
			limit = {
				exists = c:PAT
			}
			s:STATE_RIO_NEGRO = {
				add_claim = ROOT
			}
		}
		s:STATE_SOUTH_ATLANTIC_ISLANDS = {
			add_claim = ROOT
		}

		s:STATE_PATAGONIA = {
			add_claim = ROOT
		}
		
		every_country = {
			limit = {
				is_subject_type = subject_type_confederation
				overlord = ROOT
				is_player = no
			}
			save_scope_as = annex_scope
			ROOT = {
				annex = scope:annex_scope
			}
		}

	    ai_chance = {
			base = 1	
		}
    }
}

journal_flavor_argentina.11 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_argentina.11.t
	desc = journal_flavor_argentina.11.d
	flavor = journal_flavor_argentina.11.f

	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/politicians_arguing"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 1

	trigger = {
		exists = c:URU
		any_diplomatic_play = {
			initiator = root
			target = c:URU
			OR = {
				has_play_goal = conquer_state
				has_play_goal = return_state
				has_play_goal = make_dominion
				has_play_goal = make_vassal
				has_play_goal = make_puppet
			}
		}
		NOT = {
			has_variable = uruguayan_crisis
		}
	}

	immediate = {
		set_variable = {
			name = uruguayan_crisis
		}
		c:BRZ = {
			save_scope_as = brz_scope	
		}	
	}
	
	option = {
		name = journal_flavor_argentina.11.a

		c:GBR = {
			change_relations = {
				country = root
				value = -80
			}
		}

        c:BRZ = {
			change_relations = {
				country = root
				value = -80
			}	
			change_relations = {
				country = c:URU
				value = 80
			}
			trigger_event = { id = journal_flavor_argentina.12 }
		}
		    add_loyalists = {
			    value = 0.1	
			    type = soldiers	
		    }		
	    }
    }
